<inicio> ::= R_PROGRAM <lista_instr> R_ENDPROGRAM

<lista_instr> ::= <lista_instr> <instruccion>
                | <instruccion>

<instruccion> ::= <mostrar>
                | <declaracion>
                | <asignacion>
                | <declaracion_arreglo>
                | <console_print>
                | <console_column>
                | <graficacion>
                | <graficacionPie>
                | <graficacionLine>
                | <histograma>

<mostrar> ::= R_MOSTRAR PARENTESIS_A <expresion> PARENTESIS_C PUNTOYCOMA

<expresion> ::= <expresion> MAS <expresion>
              | <expresion> POR <expresion>
              | ENTERO
              | DECIMAL
              | CADENA
              | IDENTIFICADOR
              | IDENTIFICADOR_ARREGLO
              | <suma>
              | <resta>
              | <multi>
              | <division>
              | <modulo>
              | <estadistica>

<suma> ::= R_SUM PARENTESIS_A <expresion> COMMA <expresion> PARENTESIS_C

<resta> ::= R_RES PARENTESIS_A <expresion> COMMA <expresion> PARENTESIS_C

<multi> ::= R_MUL PARENTESIS_A <expresion> COMMA <expresion> PARENTESIS_C

<division> ::= R_DIV PARENTESIS_A <expresion> COMMA <expresion> PARENTESIS_C

<modulo> ::= R_MOD PARENTESIS_A <expresion> COMMA <expresion> PARENTESIS_C

<declaracion> ::= R_VAR COLLON R_DOUBLE DOUBLE_COLLON IDENTIFICADOR ASIGNACION <expresion> FIN_DECLARACION PUNTOYCOMA
                 | R_VAR COLLON R_CHAR DOUBLE_COLLON IDENTIFICADOR ASIGNACION CADENA FIN_DECLARACION PUNTOYCOMA

<asignacion> ::= IDENTIFICADOR ASIGNACION <expresion> FIN_DECLARACION

<declaracion_arreglo> ::= R_ARR_DOUBLE IDENTIFICADOR_ARREGLO ASIGNACION LISTA_VALORES FIN_DECLARACION PUNTOYCOMA
                          | R_ARR_CHAR IDENTIFICADOR_ARREGLO ASIGNACION LISTA_VALORES FIN_DECLARACION PUNTOYCOMA

<estadistica> ::= R_MEDIA PARENTESIS_A LISTA_VALORES PARENTESIS_C
                 | R_MEDIANA PARENTESIS_A LISTA_VALORES PARENTESIS_C
                 | R_MODA PARENTESIS_A LISTA_VALORES PARENTESIS_C
                 | R_VARIANZA PARENTESIS_A LISTA_VALORES PARENTESIS_C
                 | R_MAX PARENTESIS_A LISTA_VALORES PARENTESIS_C
                 | R_MIN PARENTESIS_A LISTA_VALORES PARENTESIS_C
                 | R_MEDIA PARENTESIS_A IDENTIFICADOR_ARREGLO PARENTESIS_C
                 | R_MEDIANA PARENTESIS_A IDENTIFICADOR_ARREGLO PARENTESIS_C
                 | R_MODA PARENTESIS_A IDENTIFICADOR_ARREGLO PARENTESIS_C
                 | R_VARIANZA PARENTESIS_A IDENTIFICADOR_ARREGLO PARENTESIS_C
                 | R_MAX PARENTESIS_A IDENTIFICADOR_ARREGLO PARENTESIS_C
                 | R_MIN PARENTESIS_A IDENTIFICADOR_ARREGLO PARENTESIS_C

<lista_expresiones> ::= <lista_expresiones> COMMA <expresion>
                       | <expresion>

<console_print> ::= R_CONSOLE DOUBLE_COLLON R_PRINT IGUAL <lista_expresiones> FIN_DECLARACION PUNTOYCOMA

<console_column> ::= R_CONSOLE DOUBLE_COLLON R_COLUMN IGUAL <exp_string> FLECHA IDENTIFICADOR_ARREGLO FIN_DECLARACION PUNTOYCOMA
                    | R_CONSOLE DOUBLE_COLLON R_COLUMN IGUAL <exp_string> FLECHA LISTA_VALORES FIN_DECLARACION PUNTOYCOMA

<graficacion> ::= GRAPHBAR PARENTESIS_A TITULO DOUBLE_COLLON R_CHAR IGUAL <exp_string> FIN_DECLARACION PUNTOYCOMA
                   EJEX DOUBLE_COLLON R_CHAR IGUAL LISTA_VALORES FIN_DECLARACION PUNTOYCOMA
                   EJEY DOUBLE_COLLON R_DOUBLE IGUAL LISTA_VALORES FIN_DECLARACION PUNTOYCOMA
                   TITULOX DOUBLE_COLLON R_CHAR IGUAL <exp_string> FIN_DECLARACION PUNTOYCOMA
                   TITULOY DOUBLE_COLLON R_CHAR IGUAL <exp_string> FIN_DECLARACION PUNTOYCOMA
                   EXEC_GRAPHBAR FIN_DECLARACION PUNTOYCOMA
                   PARENTESIS_C FIN_DECLARACION PUNTOYCOMA
                   | GRAPHBAR PARENTESIS_A TITULO DOUBLE_COLLON R_CHAR IGUAL <exp_string> FIN_DECLARACION PUNTOYCOMA
                   EJEX DOUBLE_COLLON R_CHAR IGUAL IDENTIFICADOR_ARREGLO FIN_DECLARACION PUNTOYCOMA
                   EJEY DOUBLE_COLLON R_DOUBLE IGUAL IDENTIFICADOR_ARREGLO FIN_DECLARACION PUNTOYCOMA
                   TITULOX DOUBLE_COLLON R_CHAR IGUAL <exp_string> FIN_DECLARACION PUNTOYCOMA
                   TITULOY DOUBLE_COLLON R_CHAR IGUAL <exp_string> FIN_DECLARACION PUNTOYCOMA
                   EXEC_GRAPHBAR FIN_DECLARACION PUNTOYCOMA
                   PARENTESIS_C FIN_DECLARACION PUNTOYCOMA

<graficacionPie> ::= GRAPHPIE PARENTESIS_A LABEL DOUBLE_COLLON R_CHAR IGUAL LISTA_VALORES FIN_DECLARACION PUNTOYCOMA
                     VALUES DOUBLE_COLLON R_DOUBLE IGUAL LISTA_VALORES FIN_DECLARACION PUNTOYCOMA
                     TITULO DOUBLE_COLLON R_CHAR IGUAL <exp_string> FIN_DECLARACION PUNTOYCOMA
                     EXEC_GRAPHPIE FIN_DECLARACION PUNTOYCOMA
                     PARENTESIS_C FIN_DECLARACION PUNTOYCOMA
                     | GRAPHPIE PARENTESIS_A LABEL DOUBLE_COLLON R_CHAR IGUAL IDENTIFICADOR_ARREGLO FIN_DECLARACION PUNTOYCOMA
                     VALUES DOUBLE_COLLON R_DOUBLE IGUAL IDENTIFICADOR_ARREGLO FIN_DECLARACION PUNTOYCOMA
                     TITULO DOUBLE_COLLON R_CHAR IGUAL <exp_string> FIN_DECLARACION PUNTOYCOMA
                     EXEC_GRAPHPIE FIN_DECLARACION PUNTOYCOMA
                     PARENTESIS_C FIN_DECLARACION PUNTOYCOMA

<graficacionLine> ::= GRAPHLINE PARENTESIS_A TITULO DOUBLE_COLLON R_CHAR IGUAL <exp_string> FIN_DECLARACION PUNTOYCOMA
                       EJEX DOUBLE_COLLON R_CHAR IGUAL LISTA_VALORES FIN_DECLARACION PUNTOYCOMA
                       EJEY DOUBLE_COLLON R_DOUBLE IGUAL LISTA_VALORES FIN_DECLARACION PUNTOYCOMA
                       TITULOX DOUBLE_COLLON R_CHAR IGUAL <exp_string> FIN_DECLARACION PUNTOYCOMA
                       TITULOY DOUBLE_COLLON R_CHAR IGUAL <exp_string> FIN_DECLARACION PUNTOYCOMA
                       EXEC_GRAPHLINE FIN_DECLARACION PUNTOYCOMA
                       PARENTESIS_C FIN_DECLARACION PUNTOYCOMA
                       | GRAPHLINE PARENTESIS_A TITULO DOUBLE_COLLON R_CHAR IGUAL <exp_string> FIN_DECLARACION PUNTOYCOMA
                       EJEX DOUBLE_COLLON R_CHAR IGUAL IDENTIFICADOR_ARREGLO FIN_DECLARACION PUNTOYCOMA
                       EJEY DOUBLE_COLLON R_DOUBLE IGUAL IDENTIFICADOR_ARREGLO FIN_DECLARACION PUNTOYCOMA
                       TITULOX DOUBLE_COLLON R_CHAR IGUAL <exp_string> FIN_DECLARACION PUNTOYCOMA
                       TITULOY DOUBLE_COLLON R_CHAR IGUAL <exp_string> FIN_DECLARACION PUNTOYCOMA
                       EXEC_GRAPHLINE FIN_DECLARACION PUNTOYCOMA
                       PARENTESIS_C FIN_DECLARACION PUNTOYCOMA
                       | GRAPHLINE PARENTESIS_A TITULO DOUBLE_COLLON R_CHAR IGUAL <exp_string> FIN_DECLARACION PUNTOYCOMA
                       EJEX DOUBLE_COLLON R_CHAR IGUAL LISTA_VALORES FIN_DECLARACION PUNTOYCOMA
                       EJEY DOUBLE_COLLON R_DOUBLE IGUAL IDENTIFICADOR_ARREGLO FIN_DECLARACION PUNTOYCOMA
                       TITULOX DOUBLE_COLLON R_CHAR IGUAL <exp_string> FIN_DECLARACION PUNTOYCOMA
                       TITULOY DOUBLE_COLLON R_CHAR IGUAL <exp_string> FIN_DECLARACION PUNTOYCOMA
                       EXEC_GRAPHLINE FIN_DECLARACION PUNTOYCOMA
                       PARENTESIS_C FIN_DECLARACION PUNTOYCOMA

<histograma> ::= HISTOGRAM PARENTESIS_A TITULO DOUBLE_COLLON R_CHAR IGUAL <exp_string> FIN_DECLARACION PUNTOYCOMA
                   VALUES DOUBLE_COLLON R_DOUBLE IGUAL LISTA_VALORES FIN_DECLARACION PUNTOYCOMA
                   EXEC_HISTOGRAM FIN_DECLARACION PUNTOYCOMA
                   PARENTESIS_C FIN_DECLARACION PUNTOYCOMA
